class bcolors:
    position = "\033[92m"
    letter = "\033[93m"
    endc = "\033[0m" 

def colorize(type,letter):
    if type == "position":
        return f"{bcolors.position}{letter}{bcolors.endc}"
    elif type == "letter":
        return f"{bcolors.letter}{letter}{bcolors.endc}"
    else:
        return letter
